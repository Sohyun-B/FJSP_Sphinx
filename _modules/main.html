def Gantt(Machines):
    """
    Machine 클래스를 이용해서 간트 차트 생성 및 생성된 차트 이미지를 png 형태로 저장합니다.
    """
    M = ['red', 'blue', 'yellow', 'orange', 'green', 'palegoldenrod', 'purple', 'pink', 'Thistle', 'Magenta',
         'SlateBlue', 'RoyalBlue', 'Cyan', 'Aqua', 'floralwhite', 'ghostwhite', 'goldenrod', 'mediumslateblue',
         'navajowhite', 'navy', 'sandybrown', 'moccasin']
    for i in range(len(Machines)):
        Machine = Machines[i]
        Start_time = Machine.O_start
        End_time = Machine.O_end
        for i_1 in range(len(End_time)):
            plt.barh(i, width=End_time[i_1] - Start_time[i_1], height=0.8, left=Start_time[i_1],
                     color=M[Machine.assigned_task[i_1][0] - 1], edgecolor='black')
            plt.text(x=Start_time[i_1] + (End_time[i_1] - Start_time[i_1]) / 2 - 0.5, y=i,
                     s=Machine.assigned_task[i_1][0])
    plt.yticks(np.arange(len(Machines) + 1), np.arange(1, len(Machines) + 2))
    plt.title('Scheduling Gantt chart')
    plt.ylabel('Machines')
    plt.xlabel('Time(min)')
    plt.savefig('Optimized_Scheduling_Gantt_Chart.png')  # Save the Gantt chart as an image file
    plt.show()

def fitness_function(cost, population_size): 
    """
    입력받은 cost(makespan)를 기반으로 각 유전자의 적합도(fitness)를 계산하고, 룰렛 휠 선택을 위한 누적 확률을 생성하는 함수.

    Args:
        cost (list): 한 세대의 염색체들의 cost(makespan) 리스트
        population_size (int): 한 세대에서 총 염색체의 수

    Returns:
        numpy.ndarray: fitness 
                        fitness[i][0]은 적합도(fitness) 점수 
                        fitness[i][1]은 룰렛 휠 선택을 위한 누적 선택 확률

    Example:
        >>> cost = [10, 20, 30]
        >>> population_size = 3
        >>> fitness = fitness_function(cost, population_size)
        >>> print(fitness)
        # 첫번쨰 열: 정규회된 적합도 값
        # 두번째 열: 누적 확률값, 룰렛 휠 선택 사용
        [[0.5   0.5 ] 
        [0.333 0.833] 
        [0.25  1.   ]]
    """
    # 적합도 계산 
    fitness = np.zeros((population_size, 2)) # 적합도 배열 초기화, fitness[0]: 각 개체의 적합도 저장, fitness[1]: 누적 확률 값 저장

    for i in range(0, fitness.shape[0]): # 각 개체의 적합도 계산
        fitness[i,0] = 1/(cost[i]) # 비용이 낮을 수록 높은 적합도 부여. 비용이 0일시 적합도는 1.
    
    # 정규화 및 누적 확률 계산
    fit_sum      = fitness[:,0].sum()
    fitness[0,1] = fitness[0,0]
    for i in range(1, fitness.shape[0]):
        fitness[i,1] = (fitness[i,0] + fitness[i-1,1])
    
    # 최종 정규화
    for i in range(0, fitness.shape[0]):
        fitness[i,1] = fitness[i,1]/fit_sum
    return fitness

def roulette_wheel(fitness):
    """
    유전 알고리즘의 룰렛 휠 선택 매커니즘을 구현한 함수
    룰렛 휠 알고리즘: 각 개체의 적합도에 비례하여 부모를 선택하는 방식. 적합도가 높은 개체가 더 자주 선택될 확률을 가짐.

    Args:
        fitness (np.ndarray):  fitness[i][0]: 각 개체의 적합도, fitness[i][1]: 누적 확률 값
    
    Returns:
        int: ix, 선택된 개체의 인덱스
    
    """ 
    ix = 0  # 선택된 개체의 인덱스를 저장할 변수

    # 0-1 사이의 난수 생성, os.urandom은 암호학적으로 안전한 난수 생성됨
    # random의 값이 룰렛의 침이 가리키는 위치. random의 값이 자신의 구간 내에 있으면 선택된 것으로 간주함.
    random = int.from_bytes(os.urandom(8), byteorder = 'big') / ((1 << 64) - 1) 
    for i in range(0, fitness.shape[0]): # 개체군의 크기만큼 반복
        if (random <= fitness[i, 1]): # 난수가 누적 확률 범위에 속하면
          ix = i # 해당 개체의 인덱스를 저장
          break # 반복 종료
    return ix # 선택된 개체의 인덱스 반환
