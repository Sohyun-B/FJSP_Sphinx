class GA():
    """
    유전자 알고리즘을 구현하는 클래스

    Attributes:
        Pop_size (int): 세대(Generation)를 이루는 염색체의 수
        Pc (float): 교치 확률
        Pm (float): 변이 확률

        Pw (float): 어떤 방식으로 변이를 수행할지 선택하는 확률 임계값
        Max_Iteration (int): 최대 반복 횟수

    def cost(self, CHS, J, Processing_time, M_num, Len, Job_constrains, Machine_constrains):
        """
        한 세대의 적합도(Cost)를 계산하는 매서드. 

        Args: 
            CHS (list): 염색체 리스트
            J (dict): 각 작업(job)의 공정(operation) 수를 나타내는 딕셔너리
            Processing_time (list): 각 작업(job)과 공정(operaiton)의 처리 시간 행렬
            M_num (int): 총 기계(machine)의 수
            Len (int): (= O_num) 총 공정(operation)의 수

        Returns:
            list: 한 세대의 염색체들의 적합도(Cost) 리스트. 여기서 적합도는 makespan

        Note:    
        """
        Cost = []
        for i in range(len(CHS)): # 각 염색체의 수만큼 반복 
            d = Decode(J, Processing_time, M_num, Job_constrains, Machine_constrains) 
            Cost.append(d.decode(CHS[i], Len)) #Cost의 리스트
        return Cost

    def avail_machine_count(self, Processing_time):
        """
        각 작업(job)의 공정(operation)에 대해 사용 가능한 기계 수(MS 값이 가질 수 있는 최대)를 리턴
        avail_machine_count[i][j]==k: i번째 job의 j번째 공정은 총 k개의 기계(machine)에서 작동시킬 수 있다.
        """
        avail_machine = []

        for i in range(len(Processing_time)):
            for j in range(len(Processing_time[i])): 
                count = 0
                for k in range(len(Processing_time[i][0])):
                    if Processing_time[i][j][k] != 9999:
                        count+=1
                avail_machine.append(count)
        return avail_machine

    def machine_cross(self, CHS1, CHS2, T0):
        """
        기계 선택의 교차 수행하는 매서드. 유전자의 MS 부분 uniform crossover로 교차. 
        Args:
            CHS1 (numpy.ndarray): 기계 교차에 선택된 유전자 1
            CHS2 (numpy.ndarray): 기계 교차에 선택된 유전자 2
            T0 (int): 총 공정(operation) 수, 교차로 선택될 기계 유전자 배열의 길이 (O_num)
        Returns: 
            tuple: 기계 교차가 된 두개의 유전자 배열
        
            
        """
        T_r = [j for j in range(T0)] # 총 공정 수 만큼 0 ~ (t0-1)까지 숫자 리스트 생성
        r = random.randint(1, 10)  # [1,10] 구간에서 정수 r을 생성
        random.shuffle(T_r) #  T_r 리스트를 무작위 셔플
        R = T_r[0:r]  # 랜덤 숫자 r의 길이인 랜덤 셔플된 리스트의 일부 R 생성 
        OS_1 = CHS1[O_num:2 * T0]
        OS_2 = CHS2[O_num:2 * T0]
        MS_1 = CHS2[0:T0]
        MS_2 = CHS1[0:T0]
        for i in R:
            K, K_2 = MS_1[i], MS_2[i]
            MS_1[i], MS_2[i] = K_2, K
        CHS1 = np.hstack((MS_1, OS_1))
        CHS2 = np.hstack((MS_2, OS_2))
        return CHS1, CHS2

    def operation_cross(self, CHS1, CHS2, T0, J_num):
        """
        공정 선택의 교차 수행하는 매서드. 유전자의 OS 부분 교차.
        CHS의 변수값이 Set1에 포함되면 순서 유지, Set1에 없으면 CHS2의 순서에 따라 매핑

        Args:
            CHS1 (numpy.ndarray): 공정 교차에 선택된 유전자 1
            CHS2 (numpy.ndarray): 공정 교차에 선택된 유전자 2
            T0 (int): 총 공정(operation) 수, 교차로 선택될 공정 유전자 배열의 길이
            J_num (int): 총 기계(machine) 수

        Returns: 
            tuple: 공정 교차가 된 두개의 유전자 배열
        
        """
        OS_1 = CHS1[T0:2 * T0]
        OS_2 = CHS2[T0:2 * T0]
        MS_1 = CHS1[0:T0]
        MS_2 = CHS2[0:T0]
        Job_list = [i for i in range(J_num)]
        random.shuffle(Job_list)
        r = random.randint(1, J_num - 1)
        Set1 = Job_list[0:r]
        new_os = list(np.zeros(T0, dtype=int))
        for k, v in enumerate(OS_1):
            if v in Set1:
                new_os[k] = v + 1
        for i in OS_2:
            if i not in Set1:
                Site = new_os.index(0)
                new_os[Site] = i + 1
        new_os = np.array([j - 1 for j in new_os])
        CHS1 = np.hstack((MS_1, new_os))
        CHS2 = np.hstack((MS_2, new_os))
        return CHS1, CHS2

    def operation_cross2(self, CHS1, CHS2, T0):
        """
        position-based crossover
        공정 선택의 교차 수행하는 매서드. 유전자의 OS 부분 교차.
        -1이나 0으로 구성된 mask를 만들어서 mask의 인덱스의 값이 -1인 경우 OS_1의 값 사용, 0인 경우 OS_2의 값을 사용하여 crossover

        Args:
            CHS1 (numpy.ndarray): 공정 교차에 선택된 유전자 1
            CHS2 (numpy.ndarray): 공정 교차에 선택된 유전자 2
            T0 (int): 총 공정(operation) 수, 교차로 선택될 공정 유전자 배열의 길이
            J_num (int): 총 기계(machine) 수

        Returns: 
            tuple: 기계 부분은 기존과 동일하고, 공정 교차가 된 두개의 유전자 배열
        """

        OS_1 = CHS1[T0:2 * T0]
        OS_2 = CHS2[T0:2 * T0]
        MS_1 = CHS1[0:T0]
        MS_2 = CHS2[0:T0]

        # 새로운 배열 초기화
        new_os = list(np.zeros(T0, dtype=int))

        # Step 2: 길이가 len(OS)이고 랜덤 숫자 r개의 값이 -1이고 나머지 값은 1인 crossover_index 배열 생성
        r = random.randint(1, T0//2)  # 랜덤 r 값 선택

        # crossover_index 생성 (r개의 1과 나머지 0)
        mask = [-1] * r + [0] * (T0 - r)
        random.shuffle(mask)  # 순서 랜덤화


        # mask가 0인 경우 CHS1의 순서를 유지
        for k, v in enumerate(OS_1):
            if mask[k] == 0:  # mask가 0인 위치는 CHS1 값 유지
                new_os[k] = v

        # mask가 -1인 경우 CHS2의 순서로 변경
        new_os_counter = Counter(new_os)
        OS_2_counter = Counter(OS_2)
        remaining_values = []
        for value in OS_2:
            if new_os_counter[value] < OS_2_counter[value]:
                remaining_values.append(value)
                new_os_counter[value] += 1

        for k in range(T0):
            if new_os[k] == -1:  # 값이 -1인 위치에만 채움
                new_os[k] = remaining_values.pop(-1)

        CHS1 = np.hstack((MS_1, new_os))
        CHS2 = np.hstack((MS_2, new_os))
        return CHS1, CHS2

    def machine_variation(self, CHS, Processing_time, T0, J):
        """
        기계 변이의 변이 수행하는 매서드. 유전자의 MS 부분 변이.
        
        Args:
            CHS (numpy.ndarray): 기계 변이에 선택된 유전자
            Processing_time (list): 공정(operation) 시간 행렬인 3차원 리스트
            T0 (int): 총 공정(operation) 수, 변이로 선택될 기계 유전자 배열의 길이
            J (dict): 각 작업(job)의 공정(operation) 수를 나타내는 딕셔너리
        Returns: 
            tuple: 기계 변이가 된 하나의 유전자 배열
            
        """
        Tr = [i_num for i_num in range(T0)]
        MS = CHS[0:T0]
        OS = CHS[T0:2 * T0]
        # 기계 선택 부분 변이 처리
        r = random.randint(1, T0 - 1)  # 변이 염색체에서 r개의 위치를 선택함.
        random.shuffle(Tr)
        T_r = Tr[0:r]
        for num in T_r:
            T_0 = [j for j in range(T0)]
            K = []
            Site = 0
            for k, v in J.items():
                K.append(T_0[Site:Site + v])
                Site += v
            for i in range(len(K)):
                if num in K[i]:
                    O_i = i
                    O_j = K[i].index(num)
                    break
            Machine_using = Processing_time[O_i][O_j] # [9999, 9999, 9999, 9999, 9999, 9999, 9999, 9999, 32, 45, 14, 20]와 같은 operation의 이용 가능한 머신
            Machine_time = [] # [ 32, 45, 14, 20]와 같이 사용 가능한 머신의 시간 사용량
            for j in Machine_using:
                if j != 9999:
                    Machine_time.append(j)
            Min_index = Machine_time.index(min(Machine_time)) # 가장 시간이 적게 걸리는 것의 인덱스
            MS[num] = Min_index
        CHS = np.hstack((MS, OS))
        return CHS

    def operation_variation(self, CHS, T0, J_num, J, O, M_num):
        """
        공정 변이의 변이 수행하는 매서드. 유전자의 OS 부분 변이.
        OS의 인덱스 job 인덱스 r개 추출하고 뽑힌 job들에 대해서 가능한 permutation 진행
        Args:
            CHS (numpy.ndarray): 공정 변이에 선택된 유전자
            T0 (int): 총 공정(operation) 수, 변이로 선택될 기계 유전자 배열의 길이
            J_num (int): 총 작업(job)의 수
            J (dict): 각 작업(job)의 공정(operation)수를 나타내는 딕셔너리
            O (list): 공정(operation) 시간 행렬
            M_num (int): 총 기계(machine)의 수
            
        Returns: 
            tuple: 공정 변이가 된 하나의 유전자 배열
            
        """
        MS = CHS[0:T0]
        OS = list(CHS[T0:2 * T0])
        r = random.randint(1, J_num - 1)
        Tr = [i for i in range(J_num)]
        random.shuffle(Tr)
        Tr = Tr[0:r] # r개의 랜덤 작업 인덱스 추출
        J_os = dict(enumerate(OS))  
        J_os = sorted(J_os.items(), key=lambda d: d[1]) #  J_os (OS였을때의 인덱스, os값)의 튜플을 가지는 리스트인데 os값으로 정렬됨
        Site = []
        for i in range(r):
            Site.append(OS.index(Tr[i]))
        A = list(itertools.permutations(Tr, r))
        A_CHS = []
        for i in range(len(A)):
            for j in range(len(A[i])):
                OS[Site[j]] = A[i][j]
            C_I = np.hstack((MS, OS))
            A_CHS.append(C_I)
        Fit = []
        for i in range(len(A_CHS)):
            d = Decode(J, O, M_num)
            Fit.append(d.decode(CHS, T0))
        return A_CHS[Fit.index(min(Fit))]
